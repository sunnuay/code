cmake_minimum_required(VERSION 4.2)
project(code)
if(CMAKE_CXX_COMPILER_ID MATCHES "Clang")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Qunused-arguments -Wno-reserved-module-identifier")
endif()
set(CMAKE_C_STANDARD 23)
set(CMAKE_CXX_STANDARD 26)
set(CMAKE_CXX_MODULE_STD ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
file(GLOB dirs "src/*")
foreach(dir ${dirs})
  if(IS_DIRECTORY "${dir}")
    get_filename_component(name "${dir}" NAME)
    file(GLOB sources CONFIGURE_DEPENDS "${dir}/*.c" "${dir}/*.cpp")
    if(sources)
      add_executable(${name} ${sources})
      if("${name}" STREQUAL "stdexec")
        find_package(stdexec CONFIG REQUIRED)
        set_target_properties(${name} PROPERTIES CXX_SCAN_FOR_MODULES OFF)
        target_link_libraries(${name} PRIVATE STDEXEC::stdexec)
      endif()
    endif()
  endif()
endforeach()
