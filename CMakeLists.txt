cmake_minimum_required(VERSION 4.2)
project(code)
set(CMAKE_CXX_STANDARD 26)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
file(GLOB entries "src/*")
foreach(entry ${entries})
  if(IS_DIRECTORY "${entry}")
    get_filename_component(name "${entry}" NAME)
    file(GLOB sources_c   CONFIGURE_DEPENDS "${entry}/*.c")
    file(GLOB sources_cpp CONFIGURE_DEPENDS "${entry}/*.cpp")
    file(GLOB sources_cppm CONFIGURE_DEPENDS "${entry}/*.cppm")
    if(sources_c OR sources_cpp OR sources_cppm)
      add_executable(${name})
      if(sources_c OR sources_cpp)
        target_sources(${name} PRIVATE ${sources_c} ${sources_cpp})
      endif()
      if(sources_cppm)
        target_sources(${name} PRIVATE FILE_SET CXX_MODULES FILES ${sources_cppm})
      endif()
      if("${name}" STREQUAL "stdexec")
        find_package(stdexec CONFIG REQUIRED)
        set_target_properties(${name} PROPERTIES CXX_SCAN_FOR_MODULES OFF)
        target_link_libraries(${name} PRIVATE STDEXEC::stdexec)
      endif()
    endif()
  endif()
endforeach()
